        // Language Toggle Functionality
        document.getElementById('languageToggle').addEventListener('click', function() {
            const arabicContents = document.querySelectorAll('.ar-content');
            const englishContents = document.querySelectorAll('.english-content');
            
            if (this.textContent === 'English') {
                // Switch to English
                arabicContents.forEach(el => el.style.display = 'none');
                englishContents.forEach(el => el.style.display = 'block');
                this.textContent = 'العربية';
                
                // Update static text
                document.getElementById('mainTitle').textContent = 'Workwise Team';
                document.getElementById('mainSubtitle').textContent = 'Exceptional job opportunities for customer service and call center professionals';
                document.getElementById('sourcingTitle').textContent = 'Candidate Sourcing Guide';
                document.getElementById('videoTitle').textContent = 'Candidate Sourcing Tutorial Videos';
                document.getElementById('jobsTitle').textContent = 'Available Jobs';
            } else {
                // Switch to Arabic
                arabicContents.forEach(el => el.style.display = 'block');
                englishContents.forEach(el => el.style.display = 'none');
                this.textContent = 'English';
                
                // Update static text
                document.getElementById('mainTitle').textContent = 'Workwise Team';
                document.getElementById('mainSubtitle').textContent = 'فرص عمل مميزة للعاملين في خدمة العملاء والمراكز الاتصالية';

                document.getElementById('jobsTitle').textContent = 'الوظائف المتاحة';
            }
        });
        
        // Copy Job Content Function
        function copyJobContent(jobId) {
            const jobElement = document.getElementById(jobId);
            const contentToCopy = jobElement.querySelector('.ar-content').style.display !== 'none' 
                ? jobElement.querySelector('.ar-content').innerText 
                : jobElement.querySelector('.english-content').innerText;
            
            navigator.clipboard.writeText(contentToCopy).then(() => {
                alert('تم نسخ محتوى الوظيفة بنجاح | Job content copied successfully');
            }).catch(err => {
                console.error('Failed to copy: ', err);
            });
        }
